  57 static inline int php_com_is_valid_object(zval *zv)
  58 {
  59         zend_class_entry *ce = Z_OBJCE_P(zv);
  60         return strcmp("com", ce->name->val) == 0 ||
  61                 strcmp("dotnet", ce->name->val) == 0 ||
  62                 strcmp("variant", ce->name->val) == 0;
  63 }
