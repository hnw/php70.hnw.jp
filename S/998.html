 390         zval value_copy;

 393

 398         }

 404                 case IS_NULL:

 406                                 value_copy = *value;

 421                                 zval_dtor(value);
 422                         }
 423                         break;
 424                 default:
 425                         php_error_docref(NULL, E_WARNING, "It is not possible to assign complex types to nodes");
 426                         break;
 427         }
