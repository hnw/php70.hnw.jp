  48 static int
  49 _strncasecmp(const char *s1, const char *s2, size_t n)
  50 {
  51         if (n != 0) {
  52                 const unsigned char *us1 = (const unsigned char *)s1,
  53                                 *us2 = (const unsigned char *)s2;
  54 
  55                 do {
  56                         if (tolower(*us1) != tolower(*us2++))
  57                                 return tolower(*us1) - tolower(*--us2);
  58                         if (*us1++ == '\0')
  59                                 break;
  60                 } while (--n != 0);
  61         }
  62         return 0;
  63 }

  65 /*
  66  * Find the first occurrence of find in s, ignore case.
  67  */
  68 char *
  69 strcasestr(const char *s, const char *find)
  70 {
  71         char c, sc;
  72         size_t len;
  73 
  74         if ((c = *find++) != 0) {
  75                 c = tolower((unsigned char)c);
  76                 len = strlen(find);
  77                 do {
  78                         do {
  79                                 if ((sc = *s++) == 0)
  80                                         return (NULL);
  81                         } while ((char)tolower((unsigned char)sc) != c);
  82                 } while (_strncasecmp(s, find, len) != 0);
  83                 s--;
  84         }
  85         return (char *)(intptr_t)(s);
  86 }
