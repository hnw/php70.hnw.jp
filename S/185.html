  97 #define RETURN_VALIDATION_FAILED        \

  99         if (flags & FILTER_NULL_ON_FAILURE) {   \

 106 #define PHP_FILTER_TRIM_DEFAULT(p, len) PHP_FILTER_TRIM_DEFAULT_EX(p, len, 1);

 108 #define PHP_FILTER_TRIM_DEFAULT_EX(p, len, return_if_empty) { \
 109         while ((len > 0)  && (*p == ' ' || *p == '\t' || *p == '\r' || *p == '\v' || *p == '\n')) { \
 110                 p++; \
 111                 len--; \
 112         } \
 113         if (len < 1 && return_if_empty) { \
 114                 RETURN_VALIDATION_FAILED \
 115         } \
 116         if (len > 0) { \
 117                 while (p[len-1] == ' ' || p[len-1] == '\t' || p[len-1] == '\r' || p[len-1] == '\v' || p[len-1] == '\n') { \
 118                         len--; \
 119                 } \
 120         } \
 121 }
