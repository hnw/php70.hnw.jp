  78 /*
  79  * koi8r => wchar
  80  */
  81 int
  82 mbfl_filt_conv_koi8r_wchar(int c, mbfl_convert_filter *filter)
  83 {
  84         int s;
  85 
  86         if (c >= 0 && c < koi8r_ucs_table_min) {
  87                 s = c;
  88         } else if (c >= koi8r_ucs_table_min && c < 0x100) {
  89                 s = koi8r_ucs_table[c - koi8r_ucs_table_min];
  90                 if (s <= 0) {
  91                         s = c;
  92                         s &= MBFL_WCSPLANE_MASK;
  93                         s |= MBFL_WCSPLANE_KOI8R;
  94                 }
  95         } else {
  96                 s = c;
  97                 s &= MBFL_WCSGROUP_MASK;
  98                 s |= MBFL_WCSGROUP_THROUGH;
  99         }
 100 
 101         CK((*filter->output_function)(s, filter->data));
 102 
 103         return c;
 104 }

 106 /*
 107  * wchar => koi8r
 108  */
 109 int
 110 mbfl_filt_conv_wchar_koi8r(int c, mbfl_convert_filter *filter)
 111 {
 112         int s, n;
 113 
 114         if (c < 0x80) {
 115                 s = c;
 116         } else {
 117                 s = -1;
 118                 n = koi8r_ucs_table_len-1;
 119                 while (n >= 0) {
 120                         if (c == koi8r_ucs_table[n]) {
 121                                 s = koi8r_ucs_table_min + n;
 122                                 break;
 123                         }
 124                         n--;
 125                 }
 126                 if (s <= 0 && (c & ~MBFL_WCSPLANE_MASK) == MBFL_WCSPLANE_KOI8R) {
 127                         s = c & MBFL_WCSPLANE_MASK;
 128                 }
 129         }
 130 
 131         if (s >= 0) {
 132                 CK((*filter->output_function)(s, filter->data));
 133         } else {
 134                 if (filter->illegal_mode != MBFL_OUTPUTFILTER_ILLEGAL_MODE_NONE) {
 135                         CK(mbfl_filt_conv_illegal_output(c, filter));
 136                 }
 137         }
 138 
 139         return c;
 140 }

 143 {
 144         if (c >= 0x80 && c < 0xff)
 145                 filter->flag = 0;
