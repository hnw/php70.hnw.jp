 116 /* This table encodes the rules for finding the end of an extended grapheme
 117 cluster. Every code point has a grapheme break property which is one of the
 118 ucp_gbXX values defined in ucp.h. The 2-dimensional table is indexed by the
 119 properties of two adjacent code points. The left property selects a word from
 120 the table, and the right property selects a bit from that word like this:
 121 
 122   ucp_gbtable[left-property] & (1 << right-property)
 123 
 124 The value is non-zero if a grapheme break is NOT permitted between the relevant
 125 two code points. The breaking rules are as follows:
 126 
 127 1. Break at the start and end of text (pretty obviously).
 128 
 129 2. Do not break between a CR and LF; otherwise, break before and   after
 130    controls.
 131 
 132 3. Do not break Hangul syllable sequences, the rules for which are:
 133 
 134     L may be followed by L, V, LV or LVT
 135     LV or V may be followed by V or T
 136     LVT or T may be followed by T
 137 
 138 4. Do not break before extending characters.
 139 
 140 The next two rules are only for extended grapheme clusters (but that's what we
 141 are implementing).
 142 
 143 5. Do not break before SpacingMarks.
 144 
 145 6. Do not break after Prepend characters.
 146 
 147 7. Otherwise, break everywhere.
 148 */
 149 
 150 const pcre_uint32 PRIV(ucp_gbtable[]) = {
 151    (1<<ucp_gbLF),                                           /*  0 CR */
 152    0,                                                       /*  1 LF */
 153    0,                                                       /*  2 Control */
 154    (1<<ucp_gbExtend)|(1<<ucp_gbSpacingMark),                /*  3 Extend */
 155    (1<<ucp_gbExtend)|(1<<ucp_gbPrepend)|                    /*  4 Prepend */
 156      (1<<ucp_gbSpacingMark)|(1<<ucp_gbL)|
 157      (1<<ucp_gbV)|(1<<ucp_gbT)|(1<<ucp_gbLV)|
 158      (1<<ucp_gbLVT)|(1<<ucp_gbOther),
 159 
 160    (1<<ucp_gbExtend)|(1<<ucp_gbSpacingMark),                /*  5 SpacingMark */
 161    (1<<ucp_gbExtend)|(1<<ucp_gbSpacingMark)|(1<<ucp_gbL)|   /*  6 L */
 162      (1<<ucp_gbL)|(1<<ucp_gbV)|(1<<ucp_gbLV)|(1<<ucp_gbLVT),
 163 
 164    (1<<ucp_gbExtend)|(1<<ucp_gbSpacingMark)|(1<<ucp_gbV)|   /*  7 V */
 165      (1<<ucp_gbT),
 166 
 167    (1<<ucp_gbExtend)|(1<<ucp_gbSpacingMark)|(1<<ucp_gbT),   /*  8 T */
 168    (1<<ucp_gbExtend)|(1<<ucp_gbSpacingMark)|(1<<ucp_gbV)|   /*  9 LV */
 169      (1<<ucp_gbT),
 170 
 171    (1<<ucp_gbExtend)|(1<<ucp_gbSpacingMark)|(1<<ucp_gbT),   /* 10 LVT */
 172    (1<<ucp_gbRegionalIndicator),                            /* 11 RegionalIndicator */
 173    (1<<ucp_gbExtend)|(1<<ucp_gbSpacingMark)                 /* 12 Other */
 174 };

 191 /* The pcre_utt[] table below translates Unicode property names into type and
 192 code values. It is searched by binary chop, so must be in collating sequence of
 193 name. Originally, the table contained pointers to the name strings in the first
 194 field of each entry. However, that leads to a large number of relocations when
 195 a shared library is dynamically loaded. A significant reduction is made by
 196 putting all the names into a single, large string and then using offsets in the
 197 table itself. Maintenance is more error-prone, but frequent changes to this
 198 data are unlikely.
 199 
 200 July 2008: There is now a script called maint/GenerateUtt.py that can be used
 201 to generate this data automatically instead of maintaining it by hand.
 202 
 203 The script was updated in March 2009 to generate a new EBCDIC-compliant
 204 version. Like all other character and string literals that are compared against
 205 the regular expression pattern, we must use STR_ macros instead of literal
 206 strings to make sure that UTF-8 support works on EBCDIC platforms. */
