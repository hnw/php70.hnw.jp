 102 static zend_always_inline zend_execute_data *phpdbg_user_execute_data(zend_execute_data *ex) {
 103         while (!ex->func || !ZEND_USER_CODE(ex->func->common.type)) {
 104                 ex = ex->prev_execute_data;
 105                 ZEND_ASSERT(ex);
 106         }
 107         return ex;
 108 }
