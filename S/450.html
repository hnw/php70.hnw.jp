  56 static int php_json_hex_to_int(char code)
  57 {
  58         if (code >= '0' && code <= '9') {
  59                 return code - '0';
  60         } else if (code >= 'A' && code <= 'F') {
  61                 return code - ('A' - 10);
  62         } else if (code >= 'a' && code <= 'f') {
  63                 return code - ('a' - 10);
  64         } else {
  65                 /* this should never happened (just to suppress compiler warning) */
  66                 return -1;
  67         }
