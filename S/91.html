  37 DBA_OPEN_FUNC(ndbm)
  38 {
  39         DBM *dbf;
  40         int gmode = 0;
  41         int filemode = 0644;
  42         dba_info *pinfo = (dba_info *) info;
  43 
  44         switch(info->mode) {
  45                 case DBA_READER:
  46                         gmode = O_RDONLY;
  47                         break;
  48                 case DBA_WRITER:
  49                         gmode = O_RDWR;
  50                         break;
  51                 case DBA_CREAT:
  52                         gmode = O_RDWR | O_CREAT;
  53                         break;
  54                 case DBA_TRUNC:
  55                         gmode = O_RDWR | O_CREAT | O_TRUNC;
  56                         break;
  57                 default:
  58                         return FAILURE; /* not possible */
  59         }
  60 
  61         if(info->argc > 0) {
  62                 convert_to_long_ex(&info->argv[0]);
  63                 filemode = Z_LVAL(info->argv[0]);
  64         }
  65 
  66         dbf = dbm_open(info->path, gmode, filemode);
  67 
  68         pinfo->dbf = dbf;
  69         return SUCCESS;
  70 }

  77 DBA_FETCH_FUNC(ndbm)
  78 {
  79         datum gval;
  80         char *new = NULL;
  81 
  82         NDBM_GKEY;
  83         gval = dbm_fetch(info->dbf, gkey);
  84         if(gval.dptr) {
  85                 if(newlen) *newlen = gval.dsize;
