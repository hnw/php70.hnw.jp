 278                                                         e = ZSTR_VAL(ctx->val.s) + ZSTR_LEN(ctx->val.s);

 280                                                 if ((e - p) && strncasecmp(p, ctx->lookup_data, (e - p))) {
 281                                                         doit = 0;
