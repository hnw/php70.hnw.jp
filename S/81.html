 202 /* {{{ php_dba_myke_key */
 203 static size_t php_dba_make_key(zval *key, char **key_str, char **key_free)
 204 {
 205         if (Z_TYPE_P(key) == IS_ARRAY) {
 206                 zval *group, *name;
 207                 HashPosition pos;
 208                 size_t len;
 209 
 210                 if (zend_hash_num_elements(Z_ARRVAL_P(key)) != 2) {
 211                         php_error_docref(NULL, E_RECOVERABLE_ERROR, "Key does not have exactly two elements: (key, name)");
 212                         return 0;
 213                 }
 214                 zend_hash_internal_pointer_reset_ex(Z_ARRVAL_P(key), &pos);
 215                 group = zend_hash_get_current_data_ex(Z_ARRVAL_P(key), &pos);
 216                 zend_hash_move_forward_ex(Z_ARRVAL_P(key), &pos);
 217                 name = zend_hash_get_current_data_ex(Z_ARRVAL_P(key), &pos);
 218                 convert_to_string_ex(group);
 219                 convert_to_string_ex(name);
 220                 if (Z_STRLEN_P(group) == 0) {
 221                         *key_str = Z_STRVAL_P(name);
 222                         *key_free = NULL;
 223                         return Z_STRLEN_P(name);
 224                 }
 225                 len = spprintf(key_str, 0, "[%s]%s", Z_STRVAL_P(group), Z_STRVAL_P(name));
 226                 *key_free = *key_str;
 227                 return len;
 228         } else {
 229                 zval tmp;
 230                 int len;
 231 
 232                 ZVAL_COPY(&tmp, key);
 233                 convert_to_string(&tmp);
 234 
 235                 len = Z_STRLEN(tmp);
 236                 if (len) {
 237                         *key_free = *key_str = estrndup(Z_STRVAL(tmp), Z_STRLEN(tmp));
 238                 }
 239                 zval_ptr_dtor(&tmp);
 240                 return len;
 241         }
 242 }

 418 /* {{{ dba_close
 419  */
 420 static void dba_close(dba_info *info)
 421 {
 422         if (info->hnd) {
 423                 info->hnd->close(info);
 424         }
 425         if (info->path) {
 426                 pefree(info->path, info->flags&DBA_PERSISTENT);
 427         }
 428         if (info->fp && info->fp != info->lock.fp) {
 429                 if (info->flags & DBA_PERSISTENT) {
 430                         php_stream_pclose(info->fp);
 431                 } else {
 432                         php_stream_close(info->fp);
 433                 }
 434         }
 435         if (info->lock.fp) {
 436                 if (info->flags & DBA_PERSISTENT) {
 437                         php_stream_pclose(info->lock.fp);
 438                 } else {
 439                         php_stream_close(info->lock.fp);
 440                 }
 441         }
 442         if (info->lock.name) {
 443                 pefree(info->lock.name, info->flags&DBA_PERSISTENT);
 444         }
 445         pefree(info, info->flags&DBA_PERSISTENT);
 446 }

 567 /* {{{ php_dba_update
 568  */
 569 static void php_dba_update(INTERNAL_FUNCTION_PARAMETERS, int mode)
 570 {
 571         size_t val_len;
 572         zval *id;
 573         dba_info *info = NULL;
 574         int ac = ZEND_NUM_ARGS();
 575         zval *key;
 576         char *val;
 577         char *key_str, *key_free;
 578         size_t key_len;
 579 
 580         if (zend_parse_parameters(ac, "zsr", &key, &val, &val_len, &id) == FAILURE) {
 581                 return;
 582         }
 583 
 584         if ((key_len = php_dba_make_key(key, &key_str, &key_free)) == 0) {
 585                 RETURN_FALSE;
 586         }
 587 
 588         DBA_FETCH_RESOURCE_WITH_ID(info, id);
 589 
 590         DBA_WRITE_CHECK_WITH_ID;
 591

 629 /* {{{ php_dba_open
 630  */
 631 static void php_dba_open(INTERNAL_FUNCTION_PARAMETERS, int persistent)
 632 {
 633         zval *args = NULL;
 634         int ac = ZEND_NUM_ARGS();
 635         dba_mode_t modenr;
 636         dba_info *info, *other;
 637         dba_handler *hptr;
 638         char *key = NULL, *error = NULL;
 639         int keylen = 0;
 640         int i;
 641         int lock_mode, lock_flag, lock_dbf = 0;
 642         char *file_mode;
 643         char mode[4], *pmode, *lock_file_mode = NULL;
 644         int persistent_flag = persistent ? STREAM_OPEN_PERSISTENT : 0;
 645         zend_string *opened_path = NULL;
 646         char *lock_name;
 647 
 648         if (ac < 2) {
 649                 WRONG_PARAM_COUNT;
 650         }
 651 
 652         /* we pass additional args to the respective handler */
 653         args = safe_emalloc(ac, sizeof(zval), 0);
 654         if (zend_get_parameters_array_ex(ac, args) != SUCCESS) {
 655                 efree(args);
 656                 WRONG_PARAM_COUNT;
 657         }
 658 
 659         /* we only take string arguments */
 660         for (i = 0; i < ac; i++) {
 661                 if (Z_TYPE(args[i]) != IS_STRING) {
 662                         convert_to_string_ex(&args[i]);
 663                 } else if (Z_REFCOUNTED(args[i])) {
 664                         Z_ADDREF(args[i]);
 665                 }
 666                 keylen += Z_STRLEN(args[i]);
 667         }
 668 
 669         if (persistent) {
 670                 zend_resource *le;
 671 
 672                 /* calculate hash */
 673                 key = safe_emalloc(keylen, 1, 1);
 674                 key[keylen] = '\0';
 675                 keylen = 0;
 676 
 677                 for(i = 0; i < ac; i++) {
 678                         memcpy(key+keylen, Z_STRVAL(args[i]), Z_STRLEN(args[i]));
 679                         keylen += Z_STRLEN(args[i]);
 680                 }
 681 
 682                 /* try to find if we already have this link in our persistent list */
 683                 if ((le = zend_hash_str_find_ptr(&EG(persistent_list), key, keylen)) != NULL) {
 684                         FREENOW;
 685 
 686                         if (le->type != le_pdb) {
 687                                 RETURN_FALSE;
 688                         }
 689 
 690                         info = (dba_info *)le->ptr;
 691 
 692                         GC_REFCOUNT(le)++;
 693                         RETURN_RES(zend_register_resource(info, le_pdb));
 694                         return;
 695                 }
 696         }
 697 
 698         if (ac==2) {
 699                 hptr = DBA_G(default_hptr);
 700                 if (!hptr) {
 701                         php_error_docref2(NULL, Z_STRVAL(args[0]), Z_STRVAL(args[1]), E_WARNING, "No default handler selected");
 702                         FREENOW;
 703                         RETURN_FALSE;
 704                 }
 705         } else {
 706                 for (hptr = handler; hptr->name && strcasecmp(hptr->name, Z_STRVAL(args[2])); hptr++);
 707         }
 708 
 709         if (!hptr->name) {
 710                 php_error_docref2(NULL, Z_STRVAL(args[0]), Z_STRVAL(args[1]), E_WARNING, "No such handler: %s", Z_STRVAL(args[2]));
 711                 FREENOW;
 712                 RETURN_FALSE;
 713         }
 714 
 715         /* Check mode: [rwnc][fl]?t?
 716          * r: Read
 717          * w: Write
 718          * n: Create/Truncate
 719          * c: Create
 720          *
 721          * d: force lock on database file
 722          * l: force lock on lck file
 723          * -: ignore locking
 724          *
 725          * t: test open database, warning if locked
 726          */
 727         strlcpy(mode, Z_STRVAL(args[1]), sizeof(mode));
 728         pmode = &mode[0];
 729         if (pmode[0] && (pmode[1]=='d' || pmode[1]=='l' || pmode[1]=='-')) { /* force lock on db file or lck file or disable locking */
 730                 switch (pmode[1]) {
 731                 case 'd':
 732                         lock_dbf = 1;
 733                         if ((hptr->flags & DBA_LOCK_ALL) == 0) {
 734                                 lock_flag = (hptr->flags & DBA_LOCK_ALL);
 735                                 break;
 736                         }
 737                         /* no break */
 738                 case 'l':
 739                         lock_flag = DBA_LOCK_ALL;
 740                         if ((hptr->flags & DBA_LOCK_ALL) == 0) {
 741                                 php_error_docref2(NULL, Z_STRVAL(args[0]), Z_STRVAL(args[1]), E_NOTICE, "Handler %s does locking internally", hptr->name);
 742                         }
 743                         break;
 744                 default:
 745                 case '-':
 746                         if ((hptr->flags & DBA_LOCK_ALL) == 0) {
 747                                 php_error_docref2(NULL, Z_STRVAL(args[0]), Z_STRVAL(args[1]), E_WARNING, "Locking cannot be disabled for handler %s", hptr->name);
 748                                 FREENOW;
 749                                 RETURN_FALSE;
 750                         }
 751                         lock_flag = 0;
 752                         break;
 753                 }
 754         } else {
 755                 lock_flag = (hptr->flags&DBA_LOCK_ALL);
 756                 lock_dbf = 1;
 757         }
 758         switch (*pmode++) {
 759                 case 'r':
 760                         modenr = DBA_READER;
 761                         lock_mode = (lock_flag & DBA_LOCK_READER) ? LOCK_SH : 0;
 762                         file_mode = "r";
 763                         break;
 764                 case 'w':
 765                         modenr = DBA_WRITER;
 766                         lock_mode = (lock_flag & DBA_LOCK_WRITER) ? LOCK_EX : 0;
 767                         file_mode = "r+b";
 768                         break;
 769                 case 'c':
 770                         modenr = DBA_CREAT;
 771                         lock_mode = (lock_flag & DBA_LOCK_CREAT) ? LOCK_EX : 0;
 772                         if (lock_mode) {
 773                                 if (lock_dbf) {
 774                                         /* the create/append check will be done on the lock
 775                                          * when the lib opens the file it is already created
 776                                          */
 777                                         file_mode = "r+b";       /* read & write, seek 0 */
 778                                         lock_file_mode = "a+b";  /* append */
 779                                 } else {
 780                                         file_mode = "a+b";       /* append */
 781                                         lock_file_mode = "w+b";  /* create/truncate */
 782                                 }
 783                         } else {
 784                                 file_mode = "a+b";
 785                         }
 786                         /* In case of the 'a+b' append mode, the handler is responsible
 787                          * to handle any rewind problems (see flatfile handler).
 788                          */
 789                         break;
 790                 case 'n':
 791                         modenr = DBA_TRUNC;
 792                         lock_mode = (lock_flag & DBA_LOCK_TRUNC) ? LOCK_EX : 0;
 793                         file_mode = "w+b";
 794                         break;
 795                 default:
 796                         php_error_docref2(NULL, Z_STRVAL(args[0]), Z_STRVAL(args[1]), E_WARNING, "Illegal DBA mode");
 797                         FREENOW;
 798                         RETURN_FALSE;
 799         }
 800         if (!lock_file_mode) {
 801                 lock_file_mode = file_mode;
 802         }
 803         if (*pmode=='d' || *pmode=='l' || *pmode=='-') {
 804                 pmode++; /* done already - skip here */
 805         }
 806         if (*pmode=='t') {
 807                 pmode++;
 808                 if (!lock_flag) {
 809                         php_error_docref2(NULL, Z_STRVAL(args[0]), Z_STRVAL(args[1]), E_WARNING, "You cannot combine modifiers - (no lock) and t (test lock)");
 810                         FREENOW;
 811                         RETURN_FALSE;
 812                 }
 813                 if (!lock_mode) {
 814                         if ((hptr->flags & DBA_LOCK_ALL) == 0) {
 815                                 php_error_docref2(NULL, Z_STRVAL(args[0]), Z_STRVAL(args[1]), E_WARNING, "Handler %s uses its own locking which doesn't support mode modifier t (test lock)", hptr->name);
 816                                 FREENOW;
 817                                 RETURN_FALSE;
 818                         } else {
 819                                 php_error_docref2(NULL, Z_STRVAL(args[0]), Z_STRVAL(args[1]), E_WARNING, "Handler %s doesn't uses locking for this mode which makes modifier t (test lock) obsolete", hptr->name);
 820                                 FREENOW;
 821                                 RETURN_FALSE;
 822                         }
 823                 } else {
 824                         lock_mode |= LOCK_NB; /* test =: non blocking */
 825                 }
 826         }
 827         if (*pmode) {
 828                 php_error_docref2(NULL, Z_STRVAL(args[0]), Z_STRVAL(args[1]), E_WARNING, "Illegal DBA mode");
 829                 FREENOW;
 830                 RETURN_FALSE;
 831         }
 832 
 833         info = pemalloc(sizeof(dba_info), persistent);
 834         memset(info, 0, sizeof(dba_info));
 835         info->path = pestrdup(Z_STRVAL(args[0]), persistent);
 836         info->mode = modenr;
 837         info->argc = ac - 3;
 838         info->argv = args + 3;
 839         info->flags = (hptr->flags & ~DBA_LOCK_ALL) | (lock_flag & DBA_LOCK_ALL) | (persistent ? DBA_PERSISTENT : 0);
 840         info->lock.mode = lock_mode;
 841 
 842         /* if any open call is a locking call:
 843          * check if we already habe a locking call open that should block this call
 844          * the problem is some systems would allow read during write
 845          */
 846         if (hptr->flags & DBA_LOCK_ALL) {
 847                 if ((other = php_dba_find(info->path)) != NULL) {
 848                         if (   ( (lock_mode&LOCK_EX)        && (other->lock.mode&(LOCK_EX|LOCK_SH)) )
 849                             || ( (other->lock.mode&LOCK_EX) && (lock_mode&(LOCK_EX|LOCK_SH))        )
 850                            ) {
 851                                 error = "Unable to establish lock (database file already open)"; /* force failure exit */
 852                         }
 853                 }
 854         }
 855 
 856         if (!error && lock_mode) {
 857                 if (lock_dbf) {
 858                         lock_name = Z_STRVAL(args[0]);
 859                 } else {
 860                         spprintf(&lock_name, 0, "%s.lck", info->path);
 861                         if (!strcmp(file_mode, "r")) {
 862                                 /* when in read only mode try to use existing .lck file first */
 863                                 /* do not log errors for .lck file while in read ony mode on .lck file */
 864                                 lock_file_mode = "rb";
 865                                 info->lock.fp = php_stream_open_wrapper(lock_name, lock_file_mode, STREAM_MUST_SEEK|IGNORE_PATH|persistent_flag, &opened_path);
 866                         }
 867                         if (!info->lock.fp) {
 868                                 /* when not in read mode or failed to open .lck file read only. now try again in create(write) mode and log errors */
 869                                 lock_file_mode = "a+b";
 870                         } else {
 871                                 if (opened_path) {
 872                                         info->lock.name = pestrndup(ZSTR_VAL(opened_path), ZSTR_LEN(opened_path), persistent);
 873                                         zend_string_release(opened_path);
 874                                 }
 875                         }
 876                 }
 877                 if (!info->lock.fp) {
 878                         info->lock.fp = php_stream_open_wrapper(lock_name, lock_file_mode, STREAM_MUST_SEEK|REPORT_ERRORS|IGNORE_PATH|persistent_flag, &opened_path);
 879                         if (info->lock.fp) {
 880                                 if (lock_dbf) {
 881                                         /* replace the path info with the real path of the opened file */
 882                                         pefree(info->path, persistent);
 883                                         info->path = pestrndup(ZSTR_VAL(opened_path), ZSTR_LEN(opened_path), persistent);
 884                                 }
 885                                 /* now store the name of the lock */
 886                                 info->lock.name = pestrndup(ZSTR_VAL(opened_path), ZSTR_LEN(opened_path), persistent);
 887                                 zend_string_release(opened_path);
 888                         }
 889                 }
 890                 if (!lock_dbf) {
 891                         efree(lock_name);
 892                 }
 893                 if (!info->lock.fp) {
 894                         dba_close(info);
 895                         /* stream operation already wrote an error message */
 896                         FREENOW;
 897                         RETURN_FALSE;
 898                 }
 899                 if (!php_stream_supports_lock(info->lock.fp)) {
 900                         error = "Stream does not support locking";
 901                 }
 902                 if (php_stream_lock(info->lock.fp, lock_mode)) {
 903                         error = "Unable to establish lock"; /* force failure exit */
 904                 }
 905         }
 906 
 907         /* centralised open stream for builtin */
 908         if (!error && (hptr->flags&DBA_STREAM_OPEN)==DBA_STREAM_OPEN) {
 909                 if (info->lock.fp && lock_dbf) {
 910                         info->fp = info->lock.fp; /* use the same stream for locking and database access */
 911                 } else {
 912                         info->fp = php_stream_open_wrapper(info->path, file_mode, STREAM_MUST_SEEK|REPORT_ERRORS|IGNORE_PATH|persistent_flag, NULL);
 913                 }
 914                 if (!info->fp) {
 915                         dba_close(info);
 916                         /* stream operation already wrote an error message */
 917                         FREENOW;
 918                         RETURN_FALSE;
 919                 }
 920                 if (hptr->flags & (DBA_NO_APPEND|DBA_CAST_AS_FD)) {
 921                         /* Needed because some systems do not allow to write to the original
 922                          * file contents with O_APPEND being set.
 923                          */
 924                         if (SUCCESS != php_stream_cast(info->fp, PHP_STREAM_AS_FD, (void*)&info->fd, 1)) {
 925                                 php_error_docref(NULL, E_WARNING, "Could not cast stream");
 926                                 dba_close(info);
 927                                 FREENOW;
 928                                 RETURN_FALSE;
 929 #ifdef F_SETFL
 930                         } else if (modenr == DBA_CREAT) {
 931                                 int flags = fcntl(info->fd, F_GETFL);
 932                                 fcntl(info->fd, F_SETFL, flags & ~O_APPEND);
 933 #endif
 934                         }
 935 
 936                 }
 937         }
 938 
 939         if (error || hptr->open(info, &error) != SUCCESS) {
 940                 dba_close(info);
 941                 php_error_docref2(NULL, Z_STRVAL(args[0]), Z_STRVAL(args[1]), E_WARNING, "Driver initialization failed for handler: %s%s%s", hptr->name, error?": ":"", error?error:"");
 942                 FREENOW;
 943                 RETURN_FALSE;
 944         }
 945 
 946         info->hnd = hptr;
 947         info->argc = 0;
 948         info->argv = NULL;
 949 
 950         if (persistent) {
 951                 zval new_le;
 952 
 953                 ZVAL_NEW_PERSISTENT_RES(&new_le, -1, info, le_pdb);
 954                 if (zend_hash_str_update(&EG(persistent_list), key, keylen, &new_le) == NULL) {
 955                         dba_close(info);
 956                         php_error_docref2(NULL, Z_STRVAL(args[0]), Z_STRVAL(args[1]), E_WARNING, "Could not register persistent resource");
 957                         FREENOW;
 958                         RETURN_FALSE;
 959                 }
 960         }
 961 
 962         RETVAL_RES(zend_register_resource(info, (persistent ? le_pdb : le_db)));
 963         FREENOW;
 964 }
 965 /* }}} */
 966 #undef FREENOW
