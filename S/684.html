 107 static int
 108 cp1251_mbc_case_fold(OnigCaseFoldType flag ARG_UNUSED,
 109              const UChar** pp, const UChar* end ARG_UNUSED, UChar* lower)
 110 {
 111   const UChar* p = *pp;
 112 
 113   *lower = ENC_CP1251_TO_LOWER_CASE(*p);
 114   (*pp)++;
 115   return 1;
 116 }

 118 static int
 119 cp1251_is_code_ctype(OnigCodePoint code, unsigned int ctype)
 120 {
 121   if (code < 256)
 122     return ENC_IS_CP1251_CTYPE(code, ctype);
 123   else
 124     return FALSE;
 125 }
