  51 ZIP_EXTERN zip_source_t *
  52 zip_source_file_create(const char *fname, zip_uint64_t start, zip_int64_t length, zip_error_t *error)
  53 {
  54     int size;
  55     wchar_t *wfname;
  56     zip_source_t *source;
  57 
  58     if (fname == NULL || length < -1) {
  59         zip_error_set(error, ZIP_ER_INVAL, 0);
  60         return NULL;
  61     }
  62 
  63     /* Convert fname from UTF-8 to Windows-friendly UTF-16. */
  64     size = MultiByteToWideChar(CP_UTF8, MB_ERR_INVALID_CHARS, fname, -1, NULL, 0);
  65     if (size == 0) {
  66         zip_error_set(error, ZIP_ER_INVAL, 0);
  67         return NULL;
  68     }
  69     if ((wfname = (wchar_t *)malloc(sizeof(wchar_t) * size)) == NULL) {
  70         zip_error_set(error, ZIP_ER_MEMORY, 0);
  71         return NULL;
  72     }
  73     MultiByteToWideChar(CP_UTF8, MB_ERR_INVALID_CHARS, fname, -1, wfname, size);
  74 
  75     source = zip_source_win32w_create(wfname, start, length, error);
  76 
  77     free(wfname);
  78     return source;
  79 }
