  43 #ifndef _GLOB_H_
  44 #define _GLOB_H_
  45 
  46 #ifndef PHP_WIN32
  47 # include <sys/cdefs.h>
  48 #endif
  49 
  50 #include "Zend/zend_stream.h"
  51 
  52 typedef struct {
  53         int gl_pathc;           /* Count of total paths so far. */
  54         int gl_matchc;          /* Count of paths matching pattern. */
  55         int gl_offs;            /* Reserved at beginning of gl_pathv. */
  56         int gl_flags;           /* Copy of flags parameter to glob. */
  57         char **gl_pathv;        /* List of paths matching pattern. */
  58                                 /* Copy of errfunc parameter to glob. */
  59         int (*gl_errfunc)(const char *, int);
  60 
  61         /*
  62          * Alternate filesystem access methods for glob; replacement
  63          * versions of closedir(3), readdir(3), opendir(3), stat(2)
  64          * and lstat(2).
  65          */
  66         void (*gl_closedir)(void *);
  67         struct dirent *(*gl_readdir)(void *);
  68         void *(*gl_opendir)(const char *);
  69         int (*gl_lstat)(const char *, zend_stat_t *);
  70         int (*gl_stat)(const char *, zend_stat_t *);
  71 } glob_t;
  72 
  73 /* Flags */
  74 #define GLOB_APPEND     0x0001  /* Append to output from previous call. */
  75 #define GLOB_DOOFFS     0x0002  /* Use gl_offs. */
  76 #define GLOB_ERR        0x0004  /* Return on error. */
  77 #define GLOB_MARK       0x0008  /* Append / to matching directories. */
  78 #define GLOB_NOCHECK    0x0010  /* Return pattern itself if nothing matches. */
  79 #define GLOB_NOSORT     0x0020  /* Don't sort. */
  80 
  81 #ifndef _POSIX_SOURCE
  82 #define GLOB_ALTDIRFUNC 0x0040  /* Use alternately specified directory funcs. */
  83 #define GLOB_BRACE      0x0080  /* Expand braces ala csh. */
  84 #define GLOB_MAGCHAR    0x0100  /* Pattern had globbing characters. */
  85 #define GLOB_NOMAGIC    0x0200  /* GLOB_NOCHECK without magic chars (csh). */
  86 #define GLOB_QUOTE      0x0400  /* Quote special chars with \. */
  87 #define GLOB_TILDE      0x0800  /* Expand tilde names from the passwd file. */
  88 #define GLOB_NOESCAPE   0x1000  /* Disable backslash escaping. */
  89 #define GLOB_LIMIT      0x2000  /* Limit pattern match output to ARG_MAX */
  90 #endif
  91 
  92 /* Error values returned by glob(3) */
  93 #define GLOB_NOSPACE    (-1)    /* Malloc call failed. */
  94 #define GLOB_ABORTED    (-2)    /* Unignored error. */
  95 #define GLOB_NOMATCH    (-3)    /* No match and GLOB_NOCHECK not set. */
  96 #define GLOB_NOSYS      (-4)    /* Function not supported. */
  97 #define GLOB_ABEND      GLOB_ABORTED
  98

 102 END_EXTERN_C()
