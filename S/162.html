  57 /*
  58  * Undo the NUL-termination kindly provided by process()
  59  * but leave at least one byte to look at
  60  */
  61 private size_t
  62 trim_nuls(const unsigned char *buf, size_t nbytes)
  63 {
  64         while (nbytes > 1 && buf[nbytes - 1] == '\0')
  65                 nbytes--;
  66 
  67         return nbytes;
  68 }

  70 protected int
  71 file_ascmagic(struct magic_set *ms, const unsigned char *buf, size_t nbytes,
  72         int text)
  73 {
  74         unichar *ubuf = NULL;
  75         size_t ulen = 0;
  76         int rv = 1;
  77 
  78         const char *code = NULL;
  79         const char *code_mime = NULL;
  80         const char *type = NULL;
  81 
  82         if (ms->flags & MAGIC_APPLE)
  83                 return 0;
  84 
  85         nbytes = trim_nuls(buf, nbytes);
  86 
  87         /* If file doesn't look like any sort of text, give up. */
  88         if (file_encoding(ms, buf, nbytes, &ubuf, &ulen, &code, &code_mime,
  89             &type) == 0)
  90                 rv = 0;
  91         else
  92                 rv = file_ascmagic_with_encoding(ms, buf, nbytes, ubuf, ulen, code,
  93                                                  type, text);
  94 
  95         free(ubuf);
  96 
  97         return rv;
  98 }

 100 protected int
 101 file_ascmagic_with_encoding(struct magic_set *ms, const unsigned char *buf,
 102     size_t nbytes, unichar *ubuf, size_t ulen, const char *code,
 103     const char *type, int text)
 104 {
 105         unsigned char *utf8_buf = NULL, *utf8_end;
 106         size_t mlen, i;
 107         int rv = -1;
 108         int mime = ms->flags & MAGIC_MIME;
 109 
 110         const char *subtype = NULL;
 111         const char *subtype_mime = NULL;
 112 
 113         int has_escapes = 0;
 114         int has_backspace = 0;
 115         int seen_cr = 0;
 116 
 117         int n_crlf = 0;
 118         int n_lf = 0;
 119         int n_cr = 0;
 120         int n_nel = 0;
 121         int executable = 0;
 122 
 123         size_t last_line_end = (size_t)-1;
 124         int has_long_lines = 0;
 125 
 126         if (ms->flags & MAGIC_APPLE)
 127                 return 0;
 128 
 129         nbytes = trim_nuls(buf, nbytes);
 130 
 131         /* If we have fewer than 2 bytes, give up. */
 132         if (nbytes <= 1) {
 133                 rv = 0;
 134                 goto done;
 135         }
 136 
 137         if (ulen > 0 && (ms->flags & MAGIC_NO_CHECK_SOFT) == 0) {
 138                 /* Convert ubuf to UTF-8 and try text soft magic */
 139                 /* malloc size is a conservative overestimate; could be
 140                    improved, or at least realloced after conversion. */
 141                 mlen = ulen * 6;
 142                 if ((utf8_buf = CAST(unsigned char *, emalloc(mlen))) == NULL) {
 143                         file_oomem(ms, mlen);
 144                         goto done;
 145                 }
 146                 if ((utf8_end = encode_utf8(utf8_buf, mlen, ubuf, ulen))
 147                     == NULL)
 148                         goto done;
 149                 if ((rv = file_softmagic(ms, utf8_buf,
 150                     (size_t)(utf8_end - utf8_buf), 0, NULL,
 151                     TEXTTEST, text)) == 0)
 152                         rv = -1;
 153         }
 154 
 155         /* Now try to discover other details about the file. */
 156         for (i = 0; i < ulen; i++) {
 157                 if (ubuf[i] == '\n') {
 158                         if (seen_cr)
 159                                 n_crlf++;
 160                         else
 161                                 n_lf++;
 162                         last_line_end = i;
 163                 } else if (seen_cr)
 164                         n_cr++;
 165 
 166                 seen_cr = (ubuf[i] == '\r');
 167                 if (seen_cr)
 168                         last_line_end = i;
 169 
 170                 if (ubuf[i] == 0x85) { /* X3.64/ECMA-43 "next line" character */
 171                         n_nel++;
 172                         last_line_end = i;
 173                 }
 174 
 175                 /* If this line is _longer_ than MAXLINELEN, remember it. */
 176                 if (i > last_line_end + MAXLINELEN)
 177                         has_long_lines = 1;
 178 
 179                 if (ubuf[i] == '\033')
 180                         has_escapes = 1;
 181                 if (ubuf[i] == '\b')
 182                         has_backspace = 1;
 183         }
 184 
 185         /* Beware, if the data has been truncated, the final CR could have
 186            been followed by a LF.  If we have HOWMANY bytes, it indicates
 187            that the data might have been truncated, probably even before
 188            this function was called. */
 189         if (seen_cr && nbytes < HOWMANY)
 190                 n_cr++;
 191 
 192         if (strcmp(type, "binary") == 0) {
 193                 rv = 0;
 194                 goto done;
 195         }
 196         if (mime) {
 197                 if (!file_printedlen(ms) && (mime & MAGIC_MIME_TYPE) != 0) {
 198                         if (subtype_mime) {
 199                                 if (file_printf(ms, "%s", subtype_mime) == -1)
 200                                         goto done;
 201                         } else {
 202                                 if (file_printf(ms, "text/plain") == -1)
 203                                         goto done;
 204                         }
 205                 }
 206         } else {
 207                 if (file_printedlen(ms)) {
 208                         switch (file_replace(ms, " text$", ", ")) {
 209                         case 0:
 210                                 switch (file_replace(ms, " text executable$",
 211                                     ", ")) {
 212                                 case 0:
 213                                         if (file_printf(ms, ", ") == -1)
 214                                                 goto done;
 215                                         break;
 216                                 case -1:
 217                                         goto done;
 218                                 default:
 219                                         executable = 1;
 220                                         break;
 221                                 }
 222                                 break;
 223                         case -1:
 224                                 goto done;
 225                         default:
 226                                 break;
 227                         }
 228                 }
 229 
 230                 if (file_printf(ms, "%s", code) == -1)
 231                         goto done;
 232 
 233                 if (subtype) {
 234                         if (file_printf(ms, " %s", subtype) == -1)
 235                                 goto done;
 236                 }
 237 
 238                 if (file_printf(ms, " %s", type) == -1)
 239                         goto done;
 240 
 241                 if (executable)
 242                         if (file_printf(ms, " executable") == -1)
 243                                 goto done;
 244 
 245                 if (has_long_lines)
 246                         if (file_printf(ms, ", with very long lines") == -1)
 247                                 goto done;
 248 
 249                 /*
 250                  * Only report line terminators if we find one other than LF,
 251                  * or if we find none at all.
 252                  */
 253                 if ((n_crlf == 0 && n_cr == 0 && n_nel == 0 && n_lf == 0) ||
 254                     (n_crlf != 0 || n_cr != 0 || n_nel != 0)) {
 255                         if (file_printf(ms, ", with") == -1)
 256                                 goto done;
 257 
 258                         if (n_crlf == 0 && n_cr == 0 && n_nel == 0 && n_lf == 0) {
 259                                 if (file_printf(ms, " no") == -1)
 260                                         goto done;
 261                         } else {
 262                                 if (n_crlf) {
 263                                         if (file_printf(ms, " CRLF") == -1)
 264                                                 goto done;
 265                                         if (n_cr || n_lf || n_nel)
 266                                                 if (file_printf(ms, ",") == -1)
 267                                                         goto done;
 268                                 }
 269                                 if (n_cr) {
 270                                         if (file_printf(ms, " CR") == -1)
 271                                                 goto done;
 272                                         if (n_lf || n_nel)
 273                                                 if (file_printf(ms, ",") == -1)
 274                                                         goto done;
 275                                 }
 276                                 if (n_lf) {
 277                                         if (file_printf(ms, " LF") == -1)
 278                                                 goto done;
 279                                         if (n_nel)
 280                                                 if (file_printf(ms, ",") == -1)
 281                                                         goto done;
 282                                 }
 283                                 if (n_nel)
 284                                         if (file_printf(ms, " NEL") == -1)
 285                                                 goto done;
 286                         }
 287 
 288                         if (file_printf(ms, " line terminators") == -1)
 289                                 goto done;
 290                 }
 291 
 292                 if (has_escapes)
 293                         if (file_printf(ms, ", with escape sequences") == -1)
 294                                 goto done;
 295                 if (has_backspace)
 296                         if (file_printf(ms, ", with overstriking") == -1)
 297                                 goto done;
 298         }
 299         rv = 1;
 300 done:
 301         if (utf8_buf)
 302                 efree(utf8_buf);
 303 
 304         return rv;
 305 }

 307 /*
 308  * Encode Unicode string as UTF-8, returning pointer to character
 309  * after end of string, or NULL if an invalid character is found.
 310  */
 311 private unsigned char *
 312 encode_utf8(unsigned char *buf, size_t len, unichar *ubuf, size_t ulen)
 313 {
 314         size_t i;
 315         unsigned char *end = buf + len;
 316 
 317         for (i = 0; i < ulen; i++) {
 318                 if (ubuf[i] <= 0x7f) {
 319                         if (end - buf < 1)
 320                                 return NULL;
 321                         *buf++ = (unsigned char)ubuf[i];
 322                 } else if (ubuf[i] <= 0x7ff) {
 323                         if (end - buf < 2)
 324                                 return NULL;
 325                         *buf++ = (unsigned char)((ubuf[i] >> 6) + 0xc0);
 326                         *buf++ = (unsigned char)((ubuf[i] & 0x3f) + 0x80);
 327                 } else if (ubuf[i] <= 0xffff) {
 328                         if (end - buf < 3)
 329                                 return NULL;
 330                         *buf++ = (unsigned char)((ubuf[i] >> 12) + 0xe0);
 331                         *buf++ = (unsigned char)(((ubuf[i] >> 6) & 0x3f) + 0x80);
 332                         *buf++ = (unsigned char)((ubuf[i] & 0x3f) + 0x80);
 333                 } else if (ubuf[i] <= 0x1fffff) {
 334                         if (end - buf < 4)
 335                                 return NULL;
 336                         *buf++ = (unsigned char)((ubuf[i] >> 18) + 0xf0);
 337                         *buf++ = (unsigned char)(((ubuf[i] >> 12) & 0x3f) + 0x80);
 338                         *buf++ = (unsigned char)(((ubuf[i] >>  6) & 0x3f) + 0x80);
 339                         *buf++ = (unsigned char)((ubuf[i] & 0x3f) + 0x80);
 340                 } else if (ubuf[i] <= 0x3ffffff) {
 341                         if (end - buf < 5)
 342                                 return NULL;
 343                         *buf++ = (unsigned char)((ubuf[i] >> 24) + 0xf8);
 344                         *buf++ = (unsigned char)(((ubuf[i] >> 18) & 0x3f) + 0x80);
 345                         *buf++ = (unsigned char)(((ubuf[i] >> 12) & 0x3f) + 0x80);
 346                         *buf++ = (unsigned char)(((ubuf[i] >>  6) & 0x3f) + 0x80);
 347                         *buf++ = (unsigned char)((ubuf[i] & 0x3f) + 0x80);
 348                 } else if (ubuf[i] <= 0x7fffffff) {
 349                         if (end - buf < 6)
 350                                 return NULL;
 351                         *buf++ = (unsigned char)((ubuf[i] >> 30) + 0xfc);
 352                         *buf++ = (unsigned char)(((ubuf[i] >> 24) & 0x3f) + 0x80);
 353                         *buf++ = (unsigned char)(((ubuf[i] >> 18) & 0x3f) + 0x80);
 354                         *buf++ = (unsigned char)(((ubuf[i] >> 12) & 0x3f) + 0x80);
 355                         *buf++ = (unsigned char)(((ubuf[i] >>  6) & 0x3f) + 0x80);
 356                         *buf++ = (unsigned char)((ubuf[i] & 0x3f) + 0x80);
 357                 } else /* Invalid character */
 358                         return NULL;
 359         }
 360 
 361         return buf;
 362 }
