  41 ZIP_EXTERN zip_t *
  42 zip_fdopen(int fd_orig, int _flags, int *zep)
  43 {
  44     int fd;
  45     FILE *fp;
  46     zip_t *za;
  47     zip_source_t *src;
  48     struct zip_error error;
  49 
  50     if (_flags < 0 || (_flags & ZIP_TRUNCATE)) {
  51         _zip_set_open_error(zep, NULL, ZIP_ER_INVAL);
  52         return  NULL;
  53     }
  54         
  55     /* We dup() here to avoid messing with the passed in fd.
  56        We could not restore it to the original state in case of error. */
  57 
  58     if ((fd=dup(fd_orig)) < 0) {
  59         _zip_set_open_error(zep, NULL, ZIP_ER_OPEN);
  60         return NULL;
  61     }
  62 
  63     if ((fp=fdopen(fd, "rb")) == NULL) {
  64         close(fd);
  65         _zip_set_open_error(zep, NULL, ZIP_ER_OPEN);
  66         return NULL;
  67     }
  68 
  69     zip_error_init(&error);
  70     if ((src = zip_source_filep_create(fp, 0, -1, &error)) == NULL) {
  71         _zip_set_open_error(zep, &error, 0);
  72         zip_error_fini(&error);
  73         return NULL;
  74     }
  75 
  76     if ((za = zip_open_from_source(src, _flags, &error)) == NULL) {
  77         _zip_set_open_error(zep, &error, 0);
  78         zip_error_fini(&error);
  79         return NULL;
  80     }
  81 
  82     zip_error_fini(&error);
  83     close(fd_orig);
  84     return za;
  85 }
