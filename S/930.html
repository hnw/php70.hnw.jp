  43 /*
  44  The default socket location is sometimes defined by configure.
  45  With libmysql `mysql_config --socket` will fill PDO_MYSQL_UNIX_ADDR
  46  and the user can use --with-mysql-sock=SOCKET which will fill
  47  PDO_MYSQL_UNIX_ADDR. If both aren't set we're using mysqlnd and use
  48  /tmp/mysql.sock as default on *nix and NULL for Windows (default
  49  named pipe name is set in mysqlnd).
  50 */
  51 #ifndef PDO_MYSQL_UNIX_ADDR
  52 # ifdef PHP_MYSQL_UNIX_SOCK_ADDR
  53 #  define PDO_MYSQL_UNIX_ADDR PHP_MYSQL_UNIX_SOCK_ADDR
  54 # else
  55 #  if !PHP_WIN32
  56 #   define PDO_MYSQL_UNIX_ADDR "/tmp/mysql.sock"
  57 #  else
  58 #   define PDO_MYSQL_UNIX_ADDR NULL
  59 #  endif
  60 # endif
  61 #endif

  64 #include "ext/mysqlnd/mysqlnd_reverse_api.h"
  65 static MYSQLND * pdo_mysql_convert_zv_to_mysqlnd(zval * zv)
  66 {
  67         if (Z_TYPE_P(zv) == IS_OBJECT && instanceof_function(Z_OBJCE_P(zv), php_pdo_get_dbh_ce())) {
  68                 pdo_dbh_t * dbh = Z_PDO_DBH_P(zv);
  69 
  70                 if (!dbh) {
  71                         php_error_docref(NULL, E_WARNING, "Failed to retrieve handle from object store");
  72                         return NULL;
  73                 }
  74 
  75                 if (dbh->driver != &pdo_mysql_driver) {
  76                         php_error_docref(NULL, E_WARNING, "Provided PDO instance is not using MySQL but %s", dbh->driver->driver_name);
  77                         return NULL;
  78                 }
  79 
  80                 return ((pdo_mysql_db_handle *)dbh->driver_data)->server;
  81         }
  82         return NULL;
  83 }

 177 {
 178         if (PDO_MYSQL_G(debug)) {

 193 static PHP_RSHUTDOWN_FUNCTION(pdo_mysql)
 194 {
 195         MYSQLND_DEBUG *dbg = PDO_MYSQL_G(dbg);
 196         PDO_DBG_ENTER("RSHUTDOWN");
 197         if (dbg) {
 198                 dbg->m->close(dbg);
 199                 dbg->m->free_handle(dbg);
 200                 PDO_MYSQL_G(dbg) = NULL;

 202 
 203         return SUCCESS;
