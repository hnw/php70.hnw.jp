  44 /* _zip_file_get_offset(za, ze):
  45    Returns the offset of the file data for entry ze.
  46 
  47    On error, fills in za->error and returns 0.
  48 */
  49 
  50 zip_uint64_t
  51 _zip_file_get_offset(const zip_t *za, zip_uint64_t idx, zip_error_t *error)
  52 {
  53     zip_uint64_t offset;
  54     zip_int32_t size;
  55 
  56     offset = za->entry[idx].orig->offset;
  57 
  58     if (zip_source_seek(za->src, (zip_int64_t)offset, SEEK_SET) < 0) {
  59         _zip_error_set_from_source(error, za->src);
  60         return 0;
  61     }
  62 
  63     /* TODO: cache? */
  64     if ((size=_zip_dirent_size(za->src, ZIP_EF_LOCAL, error)) < 0)
  65         return 0;
  66 
  67     if (offset+(zip_uint32_t)size > ZIP_INT64_MAX) {
  68         zip_error_set(error, ZIP_ER_SEEK, EFBIG);
  69         return 0;
  70     }
  71     
  72     return offset + (zip_uint32_t)size;
  73 }
