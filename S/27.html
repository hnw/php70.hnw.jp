  42 /* Here is the full subtract routine that takes care of negative numbers.
  43    N2 is subtracted from N1 and the result placed in RESULT.  SCALE_MIN
  44    is the minimum scale for the result. */
  45 
  46 void
  47 bc_sub (n1, n2, result, scale_min)
  48      bc_num n1, n2, *result;
  49      int scale_min;
  50 {
  51   bc_num diff = NULL;
  52   int cmp_res;
  53   int res_scale;
  54 
  55   if (n1->n_sign != n2->n_sign)
  56     {
  57       diff = _bc_do_add (n1, n2, scale_min);
  58       diff->n_sign = n1->n_sign;
  59     }
  60   else
  61     {
  62       /* subtraction must be done. */
  63       /* Compare magnitudes. */
  64       cmp_res = _bc_do_compare (n1, n2, FALSE, FALSE);
  65       switch (cmp_res)
  66         {
  67         case -1:
  68           /* n1 is less than n2, subtract n1 from n2. */
  69           diff = _bc_do_sub (n2, n1, scale_min);
  70           diff->n_sign = (n2->n_sign == PLUS ? MINUS : PLUS);
  71           break;
  72         case  0:
  73           /* They are equal! return zero! */
  74           res_scale = MAX (scale_min, MAX(n1->n_scale, n2->n_scale));
  75           diff = bc_new_num (1, res_scale);
  76           memset (diff->n_value, 0, res_scale+1);
  77           break;
  78         case  1:
  79           /* n2 is less than n1, subtract n2 from n1. */
  80           diff = _bc_do_sub (n1, n2, scale_min);
  81           diff->n_sign = n1->n_sign;
  82           break;
  83         }
  84     }
  85 
  86   /* Clean up and return. */
  87   bc_free_num (result);
  88   *result = diff;
  89 }
