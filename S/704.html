 107 static int
 108 koi8_mbc_case_fold(OnigCaseFoldType flag ARG_UNUSED,
 109            const UChar** pp, const UChar* end ARG_UNUSED, UChar* lower)
 110 {
 111   const UChar* p = *pp;
 112 
 113   *lower = ENC_KOI8_TO_LOWER_CASE(*p);
 114   (*pp)++;
 115   return 1;
 116 }

 137 static int
 138 koi8_is_code_ctype(OnigCodePoint code, unsigned int ctype)
 139 {
 140   if (code < 256)
 141     return ENC_IS_KOI8_CTYPE(code, ctype);
 142   else
 143     return FALSE;
 144 }
