  41 /* In some places we need to check if the number NUM is almost zero.
  42    Specifically, all but the last digit is 0 and the last digit is 1.
  43    Last digit is defined by scale. */
  44 
  45 char
  46 bc_is_near_zero (num, scale)
  47      bc_num num;
  48      int scale;
  49 {
  50   int  count;
  51   char *nptr;
  52 
  53   /* Error checking */
  54   if (scale > num->n_scale)
  55     scale = num->n_scale;
  56 
  57   /* Initialize */
  58   count = num->n_len + scale;
  59   nptr = num->n_value;
  60 
  61   /* The check */
  62   while ((count > 0) && (*nptr++ == 0)) count--;
  63 
  64   if (count != 0 && (count != 1 || *--nptr != 1))
  65     return FALSE;
  66   else
  67     return TRUE;
  68 }
