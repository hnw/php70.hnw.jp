 380 struct magic_set {
 381         struct mlist *mlist[MAGIC_SETS];        /* list of regular entries */
 382         struct cont {
 383                 size_t len;
 384                 struct level_info *li;
 385         } c;
 386         struct out {
 387                 char *buf;              /* Accumulation buffer */
 388                 char *pbuf;             /* Printable buffer */
 389         } o;
 390         uint32_t offset;
 391         int error;
 392         int flags;                      /* Control magic tests. */
 393         int event_flags;                /* Note things that happened. */
 394 #define                 EVENT_HAD_ERR           0x01
 395         const char *file;
 396         size_t line;                    /* current magic line number */
 397 
 398         /* data for searches */
 399         struct {
 400                 const char *s;          /* start of search in original source */
 401                 size_t s_len;           /* length of search region */
 402                 size_t offset;          /* starting offset in source: XXX - should this be off_t? */
 403                 size_t rm_len;          /* match length */
 404         } search;
 405 
 406         /* FIXME: Make the string dynamically allocated so that e.g.
 407            strings matched in files can be longer than MAXstring */
 408         union VALUETYPE ms_value;       /* either number or string */
 409         uint16_t indir_max;
 410         uint16_t name_max;
 411         uint16_t elf_shnum_max;
 412         uint16_t elf_phnum_max;
 413         uint16_t elf_notes_max;
 414 #define FILE_INDIR_MAX                  15
 415 #define FILE_NAME_MAX                   30
 416 #define FILE_ELF_SHNUM_MAX              32768
 417 #define FILE_ELF_PHNUM_MAX              128
 418 #define FILE_ELF_NOTES_MAX              256
 419 };
