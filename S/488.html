  74 /*
  75  * cp850 => wchar
  76  */
  77 int
  78 mbfl_filt_conv_cp850_wchar(int c, mbfl_convert_filter *filter)
  79 {
  80         int s;
  81 
  82         if (c >= 0 && c < cp850_ucs_table_min) {
  83                 s = c;
  84         } else if (c >= cp850_ucs_table_min && c < 0x100) {
  85                 s = cp850_ucs_table[c - cp850_ucs_table_min];
  86                 if (s <= 0) {
  87                         s = c;
  88                         s &= MBFL_WCSPLANE_MASK;
  89                         s |= MBFL_WCSPLANE_CP850;
  90                 }
  91         } else {
  92                 s = c;
  93                 s &= MBFL_WCSGROUP_MASK;
  94                 s |= MBFL_WCSGROUP_THROUGH;
  95         }
  96 
  97         CK((*filter->output_function)(s, filter->data));
  98 
  99         return c;
 100 }

 102 /*
 103  * wchar => cp850
 104  */
 105 int
 106 mbfl_filt_conv_wchar_cp850(int c, mbfl_convert_filter *filter)
 107 {
 108         int s, n;
 109 
 110         if (c < 0x80) {
 111                 s = c;
 112         } else {
 113                 s = -1;
 114                 n = cp850_ucs_table_len-1;
 115                 while (n >= 0) {
 116                         if (c == cp850_ucs_table[n]) {
 117                                 s = cp850_ucs_table_min + n;
 118                                 break;
 119                         }
 120                         n--;
 121                 }
 122                 if (s <= 0 && (c & ~MBFL_WCSPLANE_MASK) == MBFL_WCSPLANE_CP850) {
 123                         s = c & MBFL_WCSPLANE_MASK;
 124                 }
 125         }
 126 
 127         if (s >= 0) {
 128                 CK((*filter->output_function)(s, filter->data));
 129         } else {
 130                 if (filter->illegal_mode != MBFL_OUTPUTFILTER_ILLEGAL_MODE_NONE) {
 131                         CK(mbfl_filt_conv_illegal_output(c, filter));
 132                 }
 133         }
 134 
 135         return c;
 136 }

 138 static int mbfl_filt_ident_cp850(int c, mbfl_identify_filter *filter)
 139 {
 140         if (c >= 0x80 && c < 0xff)
 141                 filter->flag = 0;
 142         else
 143                 filter->flag = 1; /* not it */
 144         return c;
