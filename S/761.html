  57 void php_mysqli_report_index(const char *query, unsigned int status) {
  58         char index[15];
  59 
  60         if (status & SERVER_QUERY_NO_GOOD_INDEX_USED) {
  61                 strcpy(index, "Bad index");
  62         } else if (status & SERVER_QUERY_NO_INDEX_USED) {
  63                 strcpy(index, "No index");
  64         } else {
  65                 return;
  66         }
  67         php_mysqli_throw_sql_exception("00000", 0, "%s used in query/prepared statement %s", index, query);
  68 }
