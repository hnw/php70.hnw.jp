 106 static int
 107 koi8_r_mbc_case_fold(OnigCaseFoldType flag ARG_UNUSED,
 108              const UChar** pp, const UChar* end ARG_UNUSED, UChar* lower)
 109 {
 110   const UChar* p = *pp;
 111 
 112   *lower = ENC_KOI8_R_TO_LOWER_CASE(*p);
 113   (*pp)++;
 114   return 1;
 115 }

 130 static int
 131 koi8_r_is_code_ctype(OnigCodePoint code, unsigned int ctype)
 132 {
 133   if (code < 256)
 134     return ENC_IS_KOI8_R_CTYPE(code, ctype);
