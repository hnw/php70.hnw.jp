  70 static int
  71 is_code_ctype(OnigCodePoint code, unsigned int ctype)
  72 {
  73   if (code < 256)
  74     return ENC_IS_ISO_8859_8_CTYPE(code, ctype);
  75   else
  76     return FALSE;
  77 }
