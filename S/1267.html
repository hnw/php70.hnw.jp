  43 ZIP_EXTERN int
  44 zip_error_to_str(char *buf, zip_uint64_t len, int ze, int se)
  45 {
  46     const char *zs, *ss;
  47 
  48     if (ze < 0 || ze >= _zip_nerr_str)
  49         return snprintf(buf, len, "Unknown error %d", ze);
  50 
  51     zs = _zip_err_str[ze];
  52         
  53     switch (_zip_err_type[ze]) {
  54     case ZIP_ET_SYS:
  55         ss = strerror(se);
  56         break;
  57         
  58     case ZIP_ET_ZLIB:
  59         ss = zError(se);
  60         break;
  61         
  62     default:
  63         ss = NULL;
  64     }
  65 
  66     return snprintf(buf, len, "%s%s%s",
  67                     zs, (ss ? ": " : ""), (ss ? ss : ""));
  68 }
