 129 static void php_release_sysvshm(zend_resource *rsrc)
 130 {
 131         sysvshm_shm *shm_ptr = (sysvshm_shm *) rsrc->ptr;

 150 /* {{{ proto int shm_attach(int key [, int memsize [, int perm]])
 151    Creates or open a shared memory segment */
 152 PHP_FUNCTION(shm_attach)
 153 {
 154         sysvshm_shm *shm_list_ptr;
 155         char *shm_ptr;
 156         sysvshm_chunk_head *chunk_ptr;
 157         zend_long shm_key, shm_id, shm_size = php_sysvshm.init_mem, shm_flag = 0666;
 158 
 159         if (SUCCESS != zend_parse_parameters(ZEND_NUM_ARGS(), "l|ll", &shm_key, &shm_size, &shm_flag)) {
 160                 return;
 161         }
 162 
 163         if (shm_size < 1) {
 164                 php_error_docref(NULL, E_WARNING, "Segment size must be greater than zero");
 165                 RETURN_FALSE;
 166         }
 167 
 168         shm_list_ptr = (sysvshm_shm *) emalloc(sizeof(sysvshm_shm));
 169 
 170         /* get the id from a specified key or create new shared memory */
 171         if ((shm_id = shmget(shm_key, 0, 0)) < 0) {
 172                 if (shm_size < sizeof(sysvshm_chunk_head)) {
 173                         php_error_docref(NULL, E_WARNING, "failed for key 0x%px: memorysize too small", shm_key);
 174                         efree(shm_list_ptr);
 175                         RETURN_FALSE;
 176                 }
 177                 if ((shm_id = shmget(shm_key, shm_size, shm_flag | IPC_CREAT | IPC_EXCL)) < 0) {
 178                         php_error_docref(NULL, E_WARNING, "failed for key 0x%px: %s", shm_key, strerror(errno));
 179                         efree(shm_list_ptr);
 180                         RETURN_FALSE;
 181                 }
 182         }
 183 
 184         if ((shm_ptr = shmat(shm_id, NULL, 0)) == (void *) -1) {
 185                 php_error_docref(NULL, E_WARNING, "failed for key 0x%px: %s", shm_key, strerror(errno));
 186                 efree(shm_list_ptr);
 187                 RETURN_FALSE;
 188         }
 189 
 190         /* check if shm is already initialized */
 191         chunk_ptr = (sysvshm_chunk_head *) shm_ptr;
 192         if (strcmp((char*) &(chunk_ptr->magic), "PHP_SM") != 0) {
 193                 strcpy((char*) &(chunk_ptr->magic), "PHP_SM");
 194                 chunk_ptr->start = sizeof(sysvshm_chunk_head);
 195                 chunk_ptr->end = chunk_ptr->start;
 196                 chunk_ptr->total = shm_size;
 197                 chunk_ptr->free = shm_size-chunk_ptr->end;
 198         }
 199 
 200         shm_list_ptr->key = shm_key;
 201         shm_list_ptr->id = shm_id;
 202         shm_list_ptr->ptr = chunk_ptr;
 203 
 204         RETURN_RES(zend_register_resource(shm_list_ptr, php_sysvshm.le_shm));
 205 }

 223 /* {{{ proto bool shm_remove(resource shm_identifier)
 224    Removes shared memory from Unix systems */
 225 PHP_FUNCTION(shm_remove)
 226 {
 227         zval *shm_id;
 228         sysvshm_shm *shm_list_ptr;
 229 
 230         if (SUCCESS != zend_parse_parameters(ZEND_NUM_ARGS(), "r", &shm_id)) {
 231                 return;
 232         }
 233         SHM_FETCH_RESOURCE(shm_list_ptr, shm_id);
 234

 236                 php_error_docref(NULL, E_WARNING, "failed for key 0x%x, id %ld: %s", shm_list_ptr->key, Z_LVAL_P(shm_id), strerror(errno));

 238         }
 239 
 240         RETURN_TRUE;

 244 /* {{{ proto bool shm_put_var(resource shm_identifier, int variable_key, mixed variable)
 245    Inserts or updates a variable in shared memory */
 246 PHP_FUNCTION(shm_put_var)
 247 {
 248         zval *shm_id, *arg_var;
 249         int ret;
 250         zend_long shm_key;
 251         sysvshm_shm *shm_list_ptr;
 252         smart_str shm_var = {0};
 253         php_serialize_data_t var_hash;
 254 
 255         if (SUCCESS != zend_parse_parameters(ZEND_NUM_ARGS(), "rlz", &shm_id, &shm_key, &arg_var)) {
 256                 return;
 257         }
 258 
 259         /* setup string-variable and serialize */
 260         PHP_VAR_SERIALIZE_INIT(var_hash);
 261         php_var_serialize(&shm_var, arg_var, &var_hash);
 262         PHP_VAR_SERIALIZE_DESTROY(var_hash);
 263 
 264         shm_list_ptr = zend_fetch_resource(Z_RES_P(shm_id), PHP_SHM_RSRC_NAME, php_sysvshm.le_shm);
 265         if (!shm_list_ptr) {
 266                 smart_str_free(&shm_var);
 267                 RETURN_FALSE;
 268         }
 269 
 270         /* insert serialized variable into shared memory */
 271         ret = php_put_shm_data(shm_list_ptr->ptr, shm_key, shm_var.s? ZSTR_VAL(shm_var.s) : NULL, shm_var.s? ZSTR_LEN(shm_var.s) : 0);
 272 
 273         /* free string */
 274         smart_str_free(&shm_var);
 275 
 276         if (ret == -1) {
 277                 php_error_docref(NULL, E_WARNING, "not enough shared memory left");
 278                 RETURN_FALSE;
 279         }
 280         RETURN_TRUE;
 281 }

 284 /* {{{ proto mixed shm_get_var(resource id, int variable_key)
 285    Returns a variable from shared memory */
 286 PHP_FUNCTION(shm_get_var)
 287 {
 288         zval *shm_id;
 289         zend_long shm_key;
 290         sysvshm_shm *shm_list_ptr;
 291         char *shm_data;
 292         zend_long shm_varpos;
 293         sysvshm_chunk *shm_var;
 294         php_unserialize_data_t var_hash;
 295 
 296         if (SUCCESS != zend_parse_parameters(ZEND_NUM_ARGS(), "rl", &shm_id, &shm_key)) {
 297                 return;
 298         }
 299         SHM_FETCH_RESOURCE(shm_list_ptr, shm_id);
 300 
 301         /* setup string-variable and serialize */
 302         /* get serialized variable from shared memory */
 303         shm_varpos = php_check_shm_data((shm_list_ptr->ptr), shm_key);
 304 
 305         if (shm_varpos < 0) {
 306                 php_error_docref(NULL, E_WARNING, "variable key %pd doesn't exist", shm_key);
 307                 RETURN_FALSE;
 308         }
 309         shm_var = (sysvshm_chunk*) ((char *)shm_list_ptr->ptr + shm_varpos);
 310         shm_data = &shm_var->mem;
 311 
 312         PHP_VAR_UNSERIALIZE_INIT(var_hash);
 313         if (php_var_unserialize(return_value, (const unsigned char **) &shm_data, (unsigned char *) shm_data + shm_var->length, &var_hash) != 1) {
 314                 php_error_docref(NULL, E_WARNING, "variable data in shared memory is corrupted");
 315                 RETVAL_FALSE;
 316         }
 317         PHP_VAR_UNSERIALIZE_DESTROY(var_hash);
 318 }

 361 /* {{{ php_put_shm_data
 362  * inserts an ascii-string into shared memory */
 363 static int php_put_shm_data(sysvshm_chunk_head *ptr, zend_long key, const char *data, zend_long len)
 364 {
 365         sysvshm_chunk *shm_var;
 366         zend_long total_size;
 367         zend_long shm_varpos;
 368 
 369         total_size = ((zend_long) (len + sizeof(sysvshm_chunk) - 1) / sizeof(zend_long)) * sizeof(zend_long) + sizeof(zend_long); /* zend_long alligment */
 370 
 371         if ((shm_varpos = php_check_shm_data(ptr, key)) > 0) {
 372                 php_remove_shm_data(ptr, shm_varpos);
 373         }
 374 
 375         if (ptr->free < total_size) {
 376                 return -1; /* not enough memory */
 377         }
 378 
 379         shm_var = (sysvshm_chunk *) ((char *) ptr + ptr->end);
 380         shm_var->key = key;
 381         shm_var->length = len;
 382         shm_var->next = total_size;
 383         memcpy(&(shm_var->mem), data, len);
 384         ptr->end += total_size;
 385         ptr->free -= total_size;
 386         return 0;
 387 }
 388 /* }}} */

 390 /* {{{ php_check_shm_data
 391  */
 392 static zend_long php_check_shm_data(sysvshm_chunk_head *ptr, zend_long key)
 393 {
 394         zend_long pos;
 395         sysvshm_chunk *shm_var;
 396 
 397         pos = ptr->start;
 398 
 399         for (;;) {
 400                 if (pos >= ptr->end) {
 401                         return -1;
 402                 }
 403                 shm_var = (sysvshm_chunk*) ((char *) ptr + pos);
 404                 if (shm_var->key == key) {
 405                         return pos;
 406                 }
 407                 pos += shm_var->next;
 408 
 409                 if (shm_var->next <= 0 || pos < ptr->start) {
 410                         return -1;
 411                 }
 412         }
 413         return -1;
 414 }

 417 /* {{{ php_remove_shm_data
 418  */
 419 static int php_remove_shm_data(sysvshm_chunk_head *ptr, zend_long shm_varpos)
 420 {
 421         sysvshm_chunk *chunk_ptr, *next_chunk_ptr;
 422         zend_long memcpy_len;
 423 
 424         chunk_ptr = (sysvshm_chunk *) ((char *) ptr + shm_varpos);
 425         next_chunk_ptr = (sysvshm_chunk *) ((char *) ptr + shm_varpos + chunk_ptr->next);
 426 
 427         memcpy_len = ptr->end-shm_varpos - chunk_ptr->next;
 428         ptr->free += chunk_ptr->next;
 429         ptr->end -= chunk_ptr->next;
 430         if (memcpy_len > 0) {
 431                 memmove(chunk_ptr, next_chunk_ptr, memcpy_len);
 432         }
 433         return 0;
 434 }
