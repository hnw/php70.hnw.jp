  32  * "Compound Document Format", nor the "Channel Definition Format".

  60 typedef struct {
  61         uint64_t        h_magic;
  62 #define CDF_MAGIC       0xE11AB1A1E011CFD0LL
  63         uint64_t        h_uuid[2];
  64         uint16_t        h_revision;
  65         uint16_t        h_version;
  66         uint16_t        h_byte_order;
  67         uint16_t        h_sec_size_p2;
  68         uint16_t        h_short_sec_size_p2;
  69         uint8_t         h_unused0[10];
  70         uint32_t        h_num_sectors_in_sat;
  71         uint32_t        h_secid_first_directory;
  72         uint8_t         h_unused1[4];
  73         uint32_t        h_min_size_standard_stream;
  74         cdf_secid_t     h_secid_first_sector_in_short_sat;
  75         uint32_t        h_num_sectors_in_short_sat;
  76         cdf_secid_t     h_secid_first_sector_in_master_sat;
  77         uint32_t        h_num_sectors_in_master_sat;
  78         cdf_secid_t     h_master_sat[436/4];
  79 } cdf_header_t;
  80 
  81 #define CDF_SEC_SIZE(h) ((size_t)(1 << (h)->h_sec_size_p2))
  82 #define CDF_SEC_POS(h, secid) (CDF_SEC_SIZE(h) + (secid) * CDF_SEC_SIZE(h))
  83 #define CDF_SHORT_SEC_SIZE(h)   ((size_t)(1 << (h)->h_short_sec_size_p2))
  84 #define CDF_SHORT_SEC_POS(h, secid) ((secid) * CDF_SHORT_SEC_SIZE(h))
