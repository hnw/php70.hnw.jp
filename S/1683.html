  76 ZEND_API void zend_iterator_init(zend_object_iterator *iter)
  77 {
  78         zend_object_std_init(&iter->std, &zend_iterator_class_entry);
  79         iter->std.handlers = &iterator_object_handlers;
  80 }
  81 
  82 ZEND_API void zend_iterator_dtor(zend_object_iterator *iter)
  83 {
  84         if (--GC_REFCOUNT(&iter->std) > 0) {
  85                 return;
  86         }
  87 
  88         zend_objects_store_del(&iter->std);
  89 }

  91 ZEND_API zend_object_iterator* zend_iterator_unwrap(zval *array_ptr)
  92 {
  93         if (Z_TYPE_P(array_ptr) &&
  94             Z_OBJ_HT_P(array_ptr) == &iterator_object_handlers) {
  95                 return (zend_object_iterator *)Z_OBJ_P(array_ptr);
  96         }
  97         return NULL;
  98 }
