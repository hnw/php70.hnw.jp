  32 #define MAGIC_NONE              0x000000 /* No flags */
  33 #define MAGIC_DEBUG             0x000001 /* Turn on debugging */
  34 #define MAGIC_SYMLINK           0x000002 /* Follow symlinks */
  35 #define MAGIC_COMPRESS          0x000004 /* Check inside compressed files */
  36 #define MAGIC_DEVICES           0x000008 /* Look at the contents of devices */
  37 #define MAGIC_MIME_TYPE         0x000010 /* Return the MIME type */
  38 #define MAGIC_CONTINUE          0x000020 /* Return all matches */
  39 #define MAGIC_CHECK             0x000040 /* Print warnings to stderr */
  40 #define MAGIC_PRESERVE_ATIME    0x000080 /* Restore access time on exit */
  41 #define MAGIC_RAW               0x000100 /* Don't translate unprintable chars */
  42 #define MAGIC_ERROR             0x000200 /* Handle ENOENT etc as real errors */
  43 #define MAGIC_MIME_ENCODING     0x000400 /* Return the MIME encoding */
  44 #define MAGIC_MIME              (MAGIC_MIME_TYPE|MAGIC_MIME_ENCODING)
  45 #define MAGIC_APPLE             0x000800 /* Return the Apple creator and type */
  46 
  47 #define MAGIC_NO_CHECK_COMPRESS 0x001000 /* Don't check for compressed files */
  48 #define MAGIC_NO_CHECK_TAR      0x002000 /* Don't check for tar files */
  49 #define MAGIC_NO_CHECK_SOFT     0x004000 /* Don't check magic entries */
  50 #define MAGIC_NO_CHECK_APPTYPE  0x008000 /* Don't check application type */
  51 #define MAGIC_NO_CHECK_ELF      0x010000 /* Don't check for elf details */
  52 #define MAGIC_NO_CHECK_TEXT     0x020000 /* Don't check for text files */
  53 #define MAGIC_NO_CHECK_CDF      0x040000 /* Don't check for cdf files */
  54 #define MAGIC_NO_CHECK_TOKENS   0x100000 /* Don't check tokens */
  55 #define MAGIC_NO_CHECK_ENCODING 0x200000 /* Don't check text encodings */
  56 
  57 /* No built-in tests; only consult the magic file */
  58 #define MAGIC_NO_CHECK_BUILTIN  ( \
  59         MAGIC_NO_CHECK_COMPRESS | \
  60         MAGIC_NO_CHECK_TAR      | \
  61 /*      MAGIC_NO_CHECK_SOFT     | */ \
  62         MAGIC_NO_CHECK_APPTYPE  | \
  63         MAGIC_NO_CHECK_ELF      | \
  64         MAGIC_NO_CHECK_TEXT     | \
