  41 /* Convert a number NUM to a long.  The function returns only the integer
  42    part of the number.  For numbers that are too large to represent as
  43    a long, this function returns a zero.  This can be detected by checking
  44    the NUM for zero after having a zero returned. */
  45 
  46 long
  47 bc_num2long (num)
  48      bc_num num;
  49 {
  50   long val;
  51   char *nptr;
  52   int  index;
  53 
  54   /* Extract the int value, ignore the fraction. */
  55   val = 0;
  56   nptr = num->n_value;
  57   for (index=num->n_len; (index>0) && (val<=(LONG_MAX/BASE)); index--)
  58     val = val*BASE + *nptr++;
  59 
  60   /* Check for overflow.  If overflow, return zero. */
  61   if (index>0) val = 0;
  62   if (val < 0) val = 0;
  63 
  64   /* Return the value. */
  65   if (num->n_sign == PLUS)
  66     return (val);
  67   else
  68     return (-val);
  69 }
