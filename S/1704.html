  75 static zend_always_inline void zend_object_release(zend_object *obj)
  76 {
  77         if (--GC_REFCOUNT(obj) == 0) {
  78                 zend_objects_store_del(obj);
  79         } else if (UNEXPECTED(!GC_INFO(obj))) {
  80                 gc_possible_root((zend_refcounted*)obj);
  81         }
  82 }
