  37 #define MAKE_NOP(opline) do { \
  38         (opline)->op1.num = 0; \
  39         (opline)->op2.num = 0; \
  40         (opline)->result.num = 0; \
  41         (opline)->opcode = ZEND_NOP; \
  42         (opline)->op1_type =  IS_UNUSED; \
  43         (opline)->op2_type = IS_UNUSED; \
  44         (opline)->result_type = IS_UNUSED; \
  45          zend_vm_set_opcode_handler(opline); \
  46 } while (0)
