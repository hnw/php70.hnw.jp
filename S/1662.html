 132 static zend_always_inline void gc_check_possible_root(zval *z)
 133 {
 134         ZVAL_DEREF(z);
 135         if (Z_COLLECTABLE_P(z) && UNEXPECTED(!Z_GC_INFO_P(z))) {
 136                 gc_possible_root(Z_COUNTED_P(z));
 137         }
 138 }
