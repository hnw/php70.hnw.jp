 110 static zend_object *spoofchecker_clone_obj(zval *object) /* {{{ */
 111 {
 112         zend_object *new_obj_val;
 113         Spoofchecker_object *sfo, *new_sfo;
 114 
 115     sfo = Z_INTL_SPOOFCHECKER_P(object);
 116     intl_error_reset(SPOOFCHECKER_ERROR_P(sfo));
 117 
 118         new_obj_val = Spoofchecker_ce_ptr->create_object(Z_OBJCE_P(object));
 119         new_sfo = php_intl_spoofchecker_fetch_object(new_obj_val);
 120         /* clone standard parts */
 121         zend_objects_clone_members(&new_sfo->zo, &sfo->zo);
 122         /* clone internal object */
 123         new_sfo->uspoof = uspoof_clone(sfo->uspoof, SPOOFCHECKER_ERROR_CODE_P(new_sfo));
 124         if(U_FAILURE(SPOOFCHECKER_ERROR_CODE(new_sfo))) {
 125                 /* set up error in case error handler is interested */
 126                 intl_error_set( NULL, SPOOFCHECKER_ERROR_CODE(new_sfo), "Failed to clone SpoofChecker object", 0 );
 127                 Spoofchecker_objects_dtor(&new_sfo->zo); /* free new object */
 128                 zend_error(E_ERROR, "Failed to clone SpoofChecker object");
 129         }
 130         return new_obj_val;
 131 }

 134 /* {{{ spoofchecker_register_Spoofchecker_class
 135  * Initialize 'Spoofchecker' class
 136  */
 137 void spoofchecker_register_Spoofchecker_class(void)
 138 {
 139         zend_class_entry ce;
 140 
 141         /* Create and register 'Spoofchecker' class. */
 142         INIT_CLASS_ENTRY(ce, "Spoofchecker", Spoofchecker_class_functions);
 143         ce.create_object = Spoofchecker_object_create;
 144         Spoofchecker_ce_ptr = zend_register_internal_class(&ce);
 145 
 146         memcpy(&Spoofchecker_handlers, zend_get_std_object_handlers(),
 147                 sizeof Spoofchecker_handlers);
 148         Spoofchecker_handlers.offset = XtOffsetOf(Spoofchecker_object, zo);
 149         Spoofchecker_handlers.clone_obj = spoofchecker_clone_obj;
 150         Spoofchecker_handlers.dtor_obj = Spoofchecker_objects_dtor;
 151         Spoofchecker_handlers.free_obj = Spoofchecker_objects_free;
 152 
 153         if (!Spoofchecker_ce_ptr) {
 154                 zend_error(E_ERROR,
 155                         "Spoofchecker: attempt to create properties "
 156                         "on a non-registered class.");
 157                 return;
 158         }
 159 }
